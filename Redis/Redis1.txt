//--LISTAS(ELEMENTOS ORDENADOS POR ORDEN DE INSERCCION)--
//Crear e introducir valores en una lista
lpush(introduce por la izquierda) tarea(nombre de la lista) "comer"
lpush tarea "dormir"
lpush tarea "ducharse"
rpush(introduce por la derecha) tarea "estudiar"

//ver cuantos elementos tienes en la lista
llen tarea//te dice cuantos elementos tienes en la lista

//ver los elementos de la lista
lrange tarea 0 -1(llega al final del todo)

//sacar un elemento de la lista
lpop tarea //saca el primero de la lista pero le puedes decir cual quieres sacar
rpop tarea// saca el ultimo elemento de la lista


//--CONJUNTOS(SETS)(ELEMENTOS NO ORDENADOS)(SIN DUPLICADOS)--

                             //te dira los nombres del primer conjunto que no aparecen en el segundo

//Ver interseccion de ambos conjuntos

//--HASHES(ALMACENAN CONJUNTOS DE PARES CAMPO-VALOR)--

//--GEO(SIRVE PARA TRABAJAR CON COORDENADAS Y HACE CALCULOS ENTRE ELLAS)--
